# How to run Linux / MacOs
$ git clone https://github.com/73556O8/Fritzen-flask.git <br>
$ source venv/bin/activate <br>
$ pip3 install flask <br>
$ python3 app.py <br>

# How to run Windows
$ git clone https://github.com/73556O8/Fritzen-flask.git <br>
$ venv/Scripts/activate <br>
$ pip install flask <br>
$ python app.py <br>
